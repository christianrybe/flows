Flows - Kotlin's shot at concurrency


class MultipleIntOperation : Operation {
    var counter = 4;
    override fun performAsync(callback: (Int?, Throwable?) -> Unit) {

        while( counter > 0) {
            callback(1, null)
            counter--
        }
        callback(null, null)
    }

    override fun cancel() {
        println("cancelled")
    }
